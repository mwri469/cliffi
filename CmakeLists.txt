
cmake_minimum_required(VERSION 3.14)
project(cliffi LANGUAGES CXX VERSION 1.0.0)

include(FetchContent)
set(FETCHCONTENT_UPDATES_DISCONNECTED TRUE)

# Set environment variable & export
set(CMAKE_SRC_DIR "${CMAKE_SOURCE_DIR}/src")
set(ENV{CMAKE_SRC_DIR} ${CMAKE_SRC_DIR})

###############################################################
## Packages ###################################################
###############################################################

FetchContent_Declare(ftxui
    GIT_REPOSITORY https://github.com/ArthurSonzogni/ftxui
    GIT_TAG main
)
FetchContent_Declare(curl
    GIT_REPOSITORY https://github.com/curl/curl.git
    GIT_TAG curl-8_13_0
    OVERRIDE_FIND_PACKAGE
)
FetchContent_MakeAvailable(ftxui curl)

###############################################################
## Install ####################################################
###############################################################

add_library(${PROJECT_NAME}-lib
    ${CMAKE_SOURCE_DIR}/src/tickers/list.cpp
    ${CMAKE_SOURCE_DIR}/src/tickers/list.hpp
    ${CMAKE_SOURCE_DIR}/src/commands/handler.cpp
    ${CMAKE_SOURCE_DIR}/src/commands/handler.hpp
    ${CMAKE_SOURCE_DIR}/src/search/search.cpp
    ${CMAKE_SOURCE_DIR}/src/search/search.hpp
    ${CMAKE_SOURCE_DIR}/src/api/yahoo.cpp
    ${CMAKE_SOURCE_DIR}/src/api/yahoo.hpp
    ${CMAKE_SOURCE_DIR}/src/graph/chart.cpp
    ${CMAKE_SOURCE_DIR}/src/graph/chart.hpp
)
target_include_directories(${PROJECT_NAME}-lib
    PUBLIC
        ${CMAKE_SOURCE_DIR}/src
)
target_link_libraries(${PROJECT_NAME}-lib
    PRIVATE
        CURL::libcurl
        ftxui::screen
        ftxui::dom
        ftxui::component
)

add_executable(cliffi src/main.cpp)
target_link_libraries(cliffi
    PRIVATE
        ${PROJECT_NAME}-lib
        CURL::libcurl
        ftxui::screen
        ftxui::dom
        ftxui::component
)
