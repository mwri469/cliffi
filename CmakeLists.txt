
cmake_minimum_required (VERSION 3.14)

include(FetchContent)
set(FETCHCONTENT_UPDATES_DISCONNECTED TRUE)

# Set environment variable & export
set(CMAKE_SRC_DIR "${CMAKE_SOURCE_DIR}/src")
set(ENV{CMAKE_SRC_DIR} ${CMAKE_SRC_DIR})

###############################################################
## Packages ###################################################
###############################################################

FetchContent_Declare(ftxui
    GIT_REPOSITORY https://github.com/ArthurSonzogni/ftxui
    GIT_TAG main
)
FetchContent_Declare(curl
    URL https://curl.se/download/curl-8.13.0.tar.gz
    DOWNLOAD_EXTRACT_TIMESTAMP true
    OVERRIDE_FIND_PACKAGE # this one
)
FetchContent_MakeAvailable(ftxui)
FetchContent_MakeAvailable(curl)


add_library(${project_name}-lib
            ${CMAKE_SOURCE_DIR}/src/tickers/list.cpp
            ${CMAKE_SOURCE_DIR}/src/tickers/list.hpp
)

###############################################################
## Install ####################################################
###############################################################

project(diffi LANGUAGES CXX VERSION 1.0.0)
add_executable(diffi src/main.cpp)

target_link_libraries(${project_name}-lib
    PRIVATE
        CURL::libcurl
        ftxui::screen
        ftxui::dom
        ftxui::component
)
target_link_libraries(diffi
    PRIVATE
        ${project_name}-lib
        CURL::libcurl
        ftxui::screen
        ftxui::dom
        ftxui::component

)
